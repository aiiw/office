```sql
CREATE OR REPLACE PROCEDURE MY_JOB_PROCEDURE IS
BEGIN
   INSERT INTO LOG_TABLE (LOG_MESSAGE) VALUES ('Job executed at ' || SYSTIMESTAMP);
   COMMIT;
END;
```

分别以DBMS和DBMS_SCHEDULER两种方式:

```sql
DECLARE
   job_id NUMBER;
BEGIN
   DBMS_JOB.SUBMIT(
      job       => job_id,
      what      => 'BEGIN MY_JOB_PROCEDURE; END;',
      next_date => TRUNC(SYSDATE) + 1/24, -- 下一天的凌晨1点
      interval  => 'TRUNC(SYSDATE)+1+1/24' -- 每天凌晨1点
   );
   COMMIT;
END;
```

```sql
BEGIN
   DBMS_SCHEDULER.CREATE_JOB (
      job_name           => 'MY_JOB',
      job_type           => 'STORED_PROCEDURE',
      job_action         => 'MY_JOB_PROCEDURE',
      start_date         => SYSTIMESTAMP,
      repeat_interval    => 'FREQ=DAILY;BYHOUR=1;',
      enabled            => TRUE
   );
END;
```
