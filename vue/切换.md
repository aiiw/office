```vue
<script setup>
import { ref } from 'vue'


const titleClass = ref('title')
const istrue=ref('true')
function abc()
  {istrue.vaule=!istrue.vaule
  titleClass.value=istrue.vaule?'title1':'title'}
</script>

<template>
  <h1 :class="titleClass" @click="abc" >Make me red</h1> <!-- 此处添加一个动态 class 绑定 -->
</template>


<style>
.title {
  color: red;
}

.title1
  {
  color: blue;
}
</style>


```
