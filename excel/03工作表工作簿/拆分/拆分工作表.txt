Sub 按项目拆分到工作表()
    Tm = Timer
    Application.DisplayAlerts = False
    Application.ScreenUpdating = False
    For Each sh In Sheets
        If sh.Name <> Me.Name Then sh.Delete
    Next
    Set d = CreateObject("scripting.dictionary")
    a = 1 '//标题行
    b = 1 '//拆分项目所在的列
    arr = Range(Cells(a + 1, b), Cells(Rows.Count, b).End(3))
    For i = 1 To UBound(arr)
        d(arr(i, 1)) = ""
    Next
    brr = d.keys
    With Sheets(1)
        For i = 0 To UBound(brr)
            If brr(i) <> "" Then
                Sheets.Add(after:=Sheets(Sheets.Count)).Name = brr(i) '//插入工作表
                .Rows(a).AutoFilter Field:=b, Criteria1:=brr(i)
                .AutoFilter.Range.SpecialCells(12).Copy Sheets(brr(i)).Range("A1")
            End If
        Next
        .ShowAllData '消除筛选
    End With
    Me.Select
    Application.DisplayAlerts = True
    Application.ScreenUpdating = True
    MsgBox "拆分完成，用时：" & Format(Timer - Tm, "0.00秒"), 64, "拆分报表"
End Sub
