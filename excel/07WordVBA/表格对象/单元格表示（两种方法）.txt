Sub 第一种写入方法()
    Dim t As Table
        Set t = ActiveDocument.Tables(1)
        t.Cell(1, 1).Range = 1
        t.Cell(1, 2).Range = 2
        t.Cell(1, 3).Range = 3
        t.Cell(2, 1).Range = 4
        t.Cell(2, 2).Range = 5
        t.Cell(2, 3).Range = 6
End Sub


Sub 第二种写入方法()
    Set t = ActiveDocument.Tables(1).Range
    For i = 1 To t.Cells.Count
        t.Cells(i).Range = i
    Next
EndSub
