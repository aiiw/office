Function get总计价格()
    With CreateObject("vbscript.regexp")
        .Pattern = "([1-9]\d*\.?\d* 元)|(0\.\d*[1-9] 元|[1-9]\d*\.?\d*元)|(0\.\d*[1-9]元)"
        .Global = True
        .IgnoreCase = True
        .MultiLine = False
        For Each mat In .Execute("价格合计/Total price（含 13%增值税/13% VAT inclusive）：2287.5元 | 大写：贰仟贰佰捌拾柒元伍角 ")
           a = mat '正则取出符合条件的页码合集
        Next mat
    End With
End Function
